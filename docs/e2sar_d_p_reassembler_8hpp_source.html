73 // custom comparator of REHdr buffers/segments - placing outside the event queue
74 // item class just not to confuse things - needed for priority queue
75 // orders segments in the increasing order of offset
76 struct SegmentComparator {
77 bool operator() (const Segment &l, const Segment &r) const {
78 REHdr *lhdr = reinterpret_cast<REHdr*>(l.segment);
79 REHdr *rhdr = reinterpret_cast<REHdr*>(r.segment);
80 return lhdr->bufferOffset > rhdr->bufferOffset;
81 }

85 struct EventQueueItem {

94 EventQueueItem(): bytes{0}, curEnd{0}, eventNum{0}, event{nullptr}, dataId{0} {}
98 EventQueueItem(REHdr *rehdr): curEnd{0}
99 {
100 bytes = rehdr->get_bufferLength();
101 dataId = rehdr->get_dataId();
102 eventNum = rehdr->get_eventNum();
103 // use deallocates this, so we don't use a pool
104 event = new u_int8_t[rehdr->get_bufferLength()];

106 firstSegment = boost::chrono::steady_clock::now();

110 {
111 // clean up memory
112 while(!oodQueue.empty())
113 {
114 auto i = oodQueue.top();
115 rbp.free(i.segment);
116 oodQueue.pop();
117 }

139 static const size_t QSIZE{1000};
140 boost::lockfree::queue<EventQueueItem*> eventQueue{QSIZE};

143 // push event on the common event queue
144 // return 1 if event is lost, 0 on success
145 inline int enqueue(EventQueueItem* item) noexcept
146 {
147 int ret = 0;
148 if (eventQueue.push(item))
149 eventQueueDepth++;
150 else
151 ret = 1; // event lost, queue was full
152 // queue is lock free so we don't lock
153 recvThreadCond.notify_all();
154 return ret;
155 }
156
157 // pop event off the event queue
158 inline EventQueueItem* dequeue() noexcept
159 {
160 EventQueueItem *item{nullptr};
161 auto a = eventQueue.pop(item);
162 if (a)
163 {
164 eventQueueDepth--;
165 return item;
166 } else
167 return nullptr; // queue was empty
168 }
