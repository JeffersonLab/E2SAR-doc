424 // Calculate the average event rate from circular buffer
425 // note that locking lives outside this function, as needed.
426 inline EventRate_t eventRate(UnixTimeNano_t currentTimeNanos)
427 {
428 // no rate to report
429 if (eventStatsBuffer.size() == 0)
430 return 1;
431 EventNum_t eventTotal{0LL};
432 // walk the circular buffer
433 for(auto el: eventStatsBuffer)
434 {
435 // add up the events
436 eventTotal += el.eventsSinceLastSync;
437 }
438 auto timeDiff = currentTimeNanos -
439 eventStatsBuffer.begin()->lastSyncTimeNanos;
440
441 // convert to Hz and return
442 //return (eventTotal*1000000000UL)/timeDiff;
443 // this uses floating point but is more accurate at low rates
444 return std::round(static_cast<float>(eventTotal*1000000000UL)/timeDiff);
445 }
