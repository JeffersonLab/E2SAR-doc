530 // Calculate the average event rate from circular buffer
531 // note that locking lives outside this function, as needed.
532 // NOTE: this is only useful to sync messages if sequential
533 // event IDs are used. When usec timestamp is used for LB event numbers
534 // the event is constant 1 MHz
535 inline EventRate_t eventRate(UnixTimeNano_t currentTimeNanos)
536 {
537 // no rate to report
538 if (eventStatsBuffer.size() == 0)
539 return 1;
540 EventNum_t eventTotal{0LL};
541 // walk the circular buffer
542 for(auto el: eventStatsBuffer)
543 {
544 // add up the events
545 eventTotal += el.eventsSinceLastSync;
546 }
547 auto timeDiff = currentTimeNanos -
548 eventStatsBuffer.begin()->lastSyncTimeNanos;
549
550 // convert to Hz and return
551 //return (eventTotal*1000000000UL)/timeDiff;
552 // this uses floating point but is more accurate at low rates
553 return std::round(static_cast<float>(eventTotal*1000000000UL)/timeDiff);
554 }

556 // doesn't require locking as it looks at only srcId in segmenter, which
557 // never changes past initialization
558 inline void fillSyncHdr(SyncHdr *hdr, EventRate_t eventRate, UnixTimeNano_t tnano)
559 {
560 EventRate_t reportedRate{0};
561 if (not zeroRate)
562 reportedRate = eventRate;
563 EventNum_t reportedEventNum{lbEventNum};
564 if (usecAsEventNum)
565 {
566 // figure out what event number would be at this moment, don't worry about its entropy
567 auto nowT = boost::chrono::system_clock::now();
568 // Convert the time point to microseconds since the epoch
569 reportedEventNum = boost::chrono::duration_cast<boost::chrono::microseconds>(nowT.time_since_epoch()).count();
570 if (not zeroRate)
571 // 1 MHz in this case
572 reportedRate = 1000000;
573 }
574 hdr->set(eventSrcId, reportedEventNum, reportedRate, tnano);
575 }
