442 // Calculate the average event rate from circular buffer
443 // note that locking lives outside this function, as needed.
444 // NOTE: this is only useful to sync messages if sequential
445 // event IDs are used. When usec timestamp is used for LB event numbers
446 // the event is constant 1 MHz
447 inline EventRate_t eventRate(UnixTimeNano_t currentTimeNanos)
448 {
449 // no rate to report
450 if (eventStatsBuffer.size() == 0)
451 return 1;
452 EventNum_t eventTotal{0LL};
453 // walk the circular buffer
454 for(auto el: eventStatsBuffer)
455 {
456 // add up the events
457 eventTotal += el.eventsSinceLastSync;
458 }
459 auto timeDiff = currentTimeNanos -
460 eventStatsBuffer.begin()->lastSyncTimeNanos;
461
462 // convert to Hz and return
463 //return (eventTotal*1000000000UL)/timeDiff;
464 // this uses floating point but is more accurate at low rates
465 return std::round(static_cast<float>(eventTotal*1000000000UL)/timeDiff);
466 }

468 // doesn't require locking as it looks at only srcId in segmenter, which
469 // never changes past initialization
470 inline void fillSyncHdr(SyncHdr *hdr, EventRate_t eventRate, UnixTimeNano_t tnano)
471 {
472 EventRate_t reportedRate{0};
473 if (not zeroRate)
474 reportedRate = eventRate;
475 EventNum_t reportedEventNum{lbEventNum};
476 if (usecAsEventNum)
477 {
478 // figure out what event number would be at this moment, don't worry about its entropy
479 auto nowT = boost::chrono::system_clock::now();
480 // Convert the time point to microseconds since the epoch
481 reportedEventNum = boost::chrono::duration_cast<boost::chrono::microseconds>(nowT.time_since_epoch()).count();
482 if (not zeroRate)
483 // 1 MHz in this case
484 reportedRate = 1000000;
485 }
486 hdr->set(eventSrcId, reportedEventNum, reportedRate, tnano);
487 }
